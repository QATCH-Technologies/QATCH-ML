#!/bin/bash

# Create a virtual environment (optional)
python3 -m venv env
source env/bin/activate

# Install the required packages
pip install matplotlib \
            "numpy<2.0" \
            scikit-learn \
            scipy \
            xgboost \
            tensorflow \
            tsmoothie \
            pandas \
            hyperopt \
            seaborn \
            tqdm

# Optional: Freeze the installed packages to a requirements.txt file
pip freeze > requirements.txt

echo "Installation complete!"
